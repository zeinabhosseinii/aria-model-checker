// Job Dispatching with deadline monitoring.
// Correctness property: no task exceeds its deadline without being reported.

reactiveclass ProcessingUnit(4) {
    knownrebecs {
        DispatchController Controller;
    }
    statevars {
        int FinishedJobs;
    }
    ProcessingUnit() {
        FinishedJobs = 0;
    }
    msgsrv ExecuteJob(int Deadline) {
        int Duration = ?(2, 5);
        boolean DeadlineExceeded;
        if(Duration > Deadline) {
            DeadlineExceeded = true;
            Controller.StatusUpdate(DeadlineExceeded);
        } else {
            DeadlineExceeded = false;
            Controller.StatusUpdate(DeadlineExceeded);
            FinishedJobs++;
            if (FinishedJobs > 5) FinishedJobs = 1;
        }
    }
}

reactiveclass DispatchController(5) {
    knownrebecs {
        ProcessingUnit Unit1;
        ProcessingUnit Unit2;
    }
    statevars {
        int IDLE;
        int ACTIVE;
        int StatusUnit1;
        int StatusUnit2;
        int MissedDeadlines;
        int DeadlineMisses;
        int CompletedJobs;
        int[3] PendingQueue;
        int DefaultTimeout;
        int QueueCapacity;
    }
    DispatchController() {
        IDLE = 1;
        ACTIVE = 0;
        StatusUnit1 = IDLE;
        StatusUnit2 = IDLE;
        MissedDeadlines = 0;
        DeadlineMisses = 0;
        CompletedJobs = 0;
        DefaultTimeout = 2;
        PendingQueue[0] = DefaultTimeout;
        PendingQueue[1] = DefaultTimeout;
        PendingQueue[2] = DefaultTimeout;
        QueueCapacity = 3;
        self.DispatchNext();
    }
    msgsrv StatusUpdate(boolean DeadlineExceeded) {
        MissedDeadlines = 0;
        DeadlineMisses = 0;
        CompletedJobs = 0;
        if(DeadlineExceeded == true) {
            DeadlineMisses = 1;
        } else {
            CompletedJobs = 1;
        }
        if(sender == Unit1) {
            StatusUnit1 = IDLE;
        } else if(sender == Unit2) {
            StatusUnit2 = IDLE;
        }
    }
    msgsrv DispatchNext() {
        MissedDeadlines = 0;
        DeadlineMisses = 0;
        CompletedJobs = 0;
        int I = 0;
        if(StatusUnit1 == IDLE) {
            StatusUnit1 = ACTIVE;
            Unit1.ExecuteJob(PendingQueue[0]);
            I = 0;
            while(I < QueueCapacity - 1) {
                PendingQueue[I] = PendingQueue[I + 1];
                I++;
            }
            PendingQueue[QueueCapacity - 1] = DefaultTimeout;
        }
        if(StatusUnit2 == IDLE) {
            StatusUnit2 = ACTIVE;
            Unit2.ExecuteJob(PendingQueue[0]);
            I = 0;
            while(I < QueueCapacity - 1) {
                PendingQueue[I] = PendingQueue[I + 1];
                I++;
            }
            PendingQueue[QueueCapacity - 1] = DefaultTimeout;
        }
        I = 0;
        int J = 0;
        while(I < QueueCapacity) {
            PendingQueue[I]--;
            if(PendingQueue[I] == 0) {
                MissedDeadlines++;
                J = I;
                while(J < QueueCapacity - 1) {
                    PendingQueue[J] = PendingQueue[J + 1];
                    J++;
                }
                PendingQueue[QueueCapacity - 1] = DefaultTimeout;
            }
            I++;
        }
        self.DispatchNext();
    }
}

main {
    DispatchController Controller(Unit1, Unit2):();
    ProcessingUnit Unit1(Controller):();
    ProcessingUnit Unit2(Controller):();
}
